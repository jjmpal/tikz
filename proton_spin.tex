}

\newcommand{\coneback}[4][]{
  \draw[canvas is xy plane at z=#2, #1] (45-#4:#3) arc (45-#4:225+#4:#3) -- (O) --cycle;
  }
\newcommand{\conefront}[4][]{
  \draw[canvas is xy plane at z=#2, #1] (45-#4:#3) arc (45-#4:-135+#4:#3) -- (O) --cycle;
}

\newcommand{\spin}[1]{
  \ifthenelse{#1 < 0} {
    \newcommand{\mylabel}{$m_s = -\frac{1}{2}$}
    \newcommand{\topopacity}{1}
    \newcommand{\bottomopacity}{0.5}
  } {
    \newcommand{\mylabel}{$m_s = +\frac{1}{2}$}
    \newcommand{\topopacity}{1}
    \newcommand{\bottomopacity}{0.5}
  }
    \coordinate (O) at (0,0,0);
    \coneback[surface]{-3}{2}{-10}
    \draw (0,0,-4) -- (O);
    \conefront[surface]{-3}{2}{-10}
    \coneback[surface, opacity = \topopacity{}]{3}{2}{10}
    \draw[->] (O) -- (0,0,4) node[above] {$z$};
    \conefront[surface, opacity = \topopacity{}]{3}{2}{10}
    \draw[->, ultra thick] (O) -- (#1+1,#1-1,3*#1) node[above] {};
    \node at (0, 0, -4.7) {\mylabel};
  }
